
local util = require 'xlua.util'

local sync_alert = util.async_to_sync(CS.IFramework.Hotfix.Lua.MessageBox.ShowAlertBox)
local sync_confirm = util.async_to_sync(CS.IFramework.Hotfix.Lua.MessageBox.ShowConfirmBox) 

--构造alert和confirm函数
return {
    alert = function(message, title)
		 sync_alert(message, title)
    end;
	
	confirm = function(message, title)
		local ret = sync_confirm(message, title)
		return ret == true
    end;
 }
